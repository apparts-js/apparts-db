#+TITLE: @apparts/db
#+DATE: [2021-03-25 Thu]
#+AUTHOR: Philipp Uhl

Database abstraction layer for Node.js with support for multiple database engines. Currently supports *PostgreSQL*.

** Installation

#+BEGIN_SRC bash
npm install @apparts/db
#+END_SRC

You also need to install the driver for your database:

#+BEGIN_SRC bash
npm install pg
#+END_SRC

** Quick Start

#+BEGIN_SRC typescript
import { connect } from "@apparts/db";

const dbs = await connect({
  use: "postgresql",
  postgresql: {
    host: "localhost",
    port: 5432,
    user: "dbuser",
    pw: "dbpassword",
    db: "mydatabase",
  },
});

// Query a collection
const users = await dbs.collection("users").find({ active: true }).toArray();

await dbs.shutdown();
#+END_SRC

** Configuration

The ~connect~ function takes a config object specifying which database engine to use and its connection parameters.

#+BEGIN_SRC typescript
interface Config {
  use: "postgresql"; // Database engine to use
  postgresql: PGConfig;
}
#+END_SRC

*** PostgreSQL Configuration

#+BEGIN_SRC typescript
interface PGConfig {
  host: string;                    // Database host
  port: number;                    // Database port
  user: string;                    // Username
  pw: string;                      // Password
  db: string;                      // Database name
  maxPoolSize?: number;            // Max pool size (default: 10)
  connectionTimeoutMillis?: number; // Connection timeout in ms (default: 0)
  idleTimeoutMillis?: number;       // Idle timeout in ms (default: 10000)
  bigIntAsNumber?: boolean;         // Return BIGINT as number instead of string (default: false)
  idsAsBigInt?: boolean;           // Use BIGINT for auto-increment IDs (default: false)
  logs?: "errors";                 // Log mode: "errors" to log failed queries (default: no logging)
  logParams?: boolean;             // Include params in error logs (default: false)
  arrayAsJSON?: boolean;           // Store arrays as JSON strings (default: false)
  poolConfig?: PoolConfig;         // Additional pg.Pool config
}
#+END_SRC

** API

*** Database Connection (`DBS`)

#+BEGIN_SRC typescript
const dbs = await connect(config);
#+END_SRC

**** `collection(name: string): Query`

Returns a ~Query~ object for the given table/collection name.

**** `transaction<T>(fn: (t: Transaction) => Promise<T>): Promise<T>`

Executes a function within a database transaction. Automatically commits on success or rolls back on error.

#+BEGIN_SRC typescript
const result = await dbs.transaction(async (t) => {
  await t.collection("users").insert([{ name: "Alice" }]);
  await t.collection("orders").insert([{ userId: 1, total: 100 }]);
  return "success";
});
#+END_SRC

**** `raw<T>(query: string, params?: any[]): Promise<Result<T>>`

Execute a raw SQL query.

#+BEGIN_SRC typescript
const result = await dbs.raw("SELECT * FROM users WHERE active = $1", [true]);
#+END_SRC

**** `createCollection(name, indexes, fields, prefix?)`

Create a new table programmatically.

#+BEGIN_SRC typescript
await dbs.createCollection(
  "users",
  [
    { name: "id", key: ["id"] },           // Primary key
    { name: "email", unique: true },        // Unique constraint
    { name: "groupId", foreign: { table: "groups", field: "id" } }, // Foreign key
  ],
  [
    { name: "id", type: "serial", notNull: true },
    { name: "email", type: "text", notNull: true },
    { name: "groupId", type: "integer" },
    { name: "createdAt", type: "bigint" },
  ]
);
#+END_SRC

**** `shutdown(): Promise<void>`

Closes the database connection pool.

*** Query

Query objects are created via ~dbs.collection("tableName")~.

**** `find(params, pagination?): this`

Find rows matching the given parameters.

The second argument accepts either a ~Pagination~ object or positional arguments
(~limit~, ~offset~, ~order~) for backward compatibility.

#+BEGIN_SRC typescript
// Simple equality
await dbs.collection("users").find({ role: "admin" }).toArray();

// With Pagination object (recommended)
await dbs.collection("users")
  .find({ active: true }, { limit: 10, offset: 0, order: [{ key: "createdAt", dir: "DESC" }] })
  .toArray();

// Backward-compatible positional args
await dbs.collection("users")
  .find({ active: true }, 10, 0, [{ key: "createdAt", dir: "DESC" }])
  .toArray();
#+END_SRC

**** `findById(id, pagination?): this`

Alias for ~find~.

**** `findByIds(ids, pagination?): this`

Find rows where IDs match any value in the given arrays.

#+BEGIN_SRC typescript
await dbs.collection("users").findByIds({ id: [1, 2, 3] }).toArray();
await dbs.collection("users").findByIds({ id: [1, 2, 3] }, { limit: 2 }).toArray();
#+END_SRC

**** `toArray<T>(): Promise<T[]>`

Execute the query and return all matching rows.

**** `count(): Promise<number>`

Return the count of matching rows.

#+BEGIN_SRC typescript
const count = await dbs.collection("users").find({ active: true }).count();
#+END_SRC

**** `insert(content, returning?): Promise<Record<string, Id>[]>`

Insert one or more rows.

#+BEGIN_SRC typescript
const inserted = await dbs.collection("users").insert(
  [{ name: "Alice", email: "alice@example.com" }],
  ["id"] // Return the generated IDs
);
#+END_SRC

**** `update(filter, content): Promise<Result<T>>`

Update all rows matching the filter.

#+BEGIN_SRC typescript
await dbs.collection("users").update(
  { id: 1 },
  { name: "Alice Updated" }
);
#+END_SRC

**** `updateOne(filter, content): Promise<Result<T>>`

Alias for ~update~.

**** `remove(params): Promise<Result<T>>`

Delete rows matching the parameters.

#+BEGIN_SRC typescript
await dbs.collection("users").remove({ id: 1 });
#+END_SRC

**** `drop(): Promise<void>`

Drop the table.

*** Order

The ~order~ parameter is given as an array of objects. The order is
established using the first array element. If two elements are equal
according to that order, the next array element is used for ordering
(and so on).

The order array takes this form:

#+BEGIN_SRC typescript
[{ key: "<field>", dir: "ASC" | "DESC" }, ...]
#+END_SRC

The object can contain these keys:
- ~key~ :: (required) The field that should be ordered.
- ~dir~ :: (required) The direction: ~"ASC"~ for ascending or ~"DESC"~ for descending.
- ~path~ :: (optional) If ~key~ describes a JSON field, ~path~ can be used
  to specify by which element within a JSON object should be ordered.
  Path is then an array of strings (keys) that define the path within
  the nested JSON object.

#+BEGIN_SRC typescript
// Order by a nested JSON property
await dbs.collection("users")
  .find({}, { limit: 10, order: [{ key: "metadata", dir: "ASC", path: ["age"] }] })
  .toArray();
#+END_SRC

*** Pagination

The ~find~, ~findById~, and ~findByIds~ methods accept an optional ~Pagination~ object
as the second argument. The ~cursor~ field is reserved for future keyset pagination.

#+BEGIN_SRC typescript
type PaginationCursor = string | Record<string, unknown>;

type Pagination = {
  limit?: number;
  offset?: number;
  order?: Order;
  cursor?: PaginationCursor;
};
#+END_SRC

*** Result

~update~, ~updateOne~, and ~remove~ return a ~Result<T>~ object containing the affected
rows and the count.

#+BEGIN_SRC typescript
type Result<T> = {
  rows: T[];
  rowCount?: number | null;
  nextCursor?: PaginationCursor;
};
#+END_SRC

*** Filters and Operators

The ~find~ methods accept a ~Params~ object where values can be primitives or filter operators. The value is either a direct value or an object that has ~op~ and ~val~ keys.

Supported operators:
- ~in~ :: ~val~ must be an array of possible values. The filter accepts rows where the field appears in the ~val~ array.
- ~notin~ :: ~val~ must be an array of values. The filter accepts rows where the field does *not* appear in the ~val~ array.
- ~of~ :: JSONB path query. ~val~ is an object with:
  - ~path~ — an array of keys for the nested JSON property.
  - ~value~ — either a direct value to compare, or another filter operator.
  - ~cast~ (optional) — cast the JSON value before comparison. Can be ~"number"~ or ~"boolean"~. PostgreSQL returns strings for JSON values, so numeric comparisons require ~cast: "number"~.
- ~oftype~ :: Checks the type of a JSONB property. Requires a JSONB column. ~val~ is an object with:
  - ~path~ — an array of keys for the nested JSON property.
  - ~value~ — one of ~"object"~, ~"array"~, ~"string"~, ~"number"~, ~"boolean"~, or ~"null"~.
- ~exists~ :: Checks if an optional field is ~null~ or not. If ~val~ is ~true~, the field must be not ~null~. If ~val~ is ~false~, the field must be ~null~.
- ~lte~ / ~lt~ / ~gte~ / ~gt~ :: Comparison operators. ~val~ must be a number.
- ~like~ / ~ilike~ :: Pattern matching with SQL ~LIKE~. ~ilike~ is case-insensitive. Use ~%~ as a wildcard.
- ~and~ :: Combines multiple filters with logical AND. ~val~ must be an array of filter objects.

#+BEGIN_SRC typescript
type Filter =
  | { op: "and"; val: Filter[] }
  | { op: "in"; val: (string | number | boolean | null)[] }
  | { op: "notin"; val: (string | number | boolean | null)[] }
  | { op: "of"; val: { path: string[]; value: Filter | primitive; cast?: "number" | "boolean" } }
  | { op: "oftype"; val: { path: string[]; value: "object" | "array" | "string" | "number" | "boolean" | "null" } }
  | { op: "exists"; val: boolean }
  | { op: "lte" | "lt" | "gte" | "gt"; val: number }
  | { op: "like" | "ilike"; val: string };
#+END_SRC

**** Examples

#+BEGIN_SRC typescript
// Range query
await dbs.collection("products").find({
  price: { op: "gte", val: 10 }
}).toArray();

// IN operator
await dbs.collection("users").find({
  role: { op: "in", val: ["admin", "moderator"] }
}).toArray();

// NOT IN operator
await dbs.collection("users").find({
  role: { op: "notin", val: ["banned"] }
}).toArray();

// JSONB query (PostgreSQL)
await dbs.collection("users").find({
  metadata: { op: "of", val: { path: ["age"], value: 30 } }
}).toArray();

// JSONB query with cast for numeric comparison
await dbs.collection("users").find({
  metadata: { op: "of", val: { path: ["age"], value: { op: "gte", val: 18 }, cast: "number" } }
}).toArray();

// Pattern matching (case-insensitive)
await dbs.collection("users").find({
  name: { op: "ilike", val: "%alice%" }
}).toArray();

// Exists check
await dbs.collection("users").find({
  deletedAt: { op: "exists", val: false }
}).toArray();

// Combined conditions
await dbs.collection("users").find({
  age: { op: "and", val: [
    { op: "gte", val: 18 },
    { op: "lt", val: 65 }
  ]}
}).toArray();
#+END_SRC

**** Filter Grammar

The full filter syntax:

#+BEGIN_SRC js
const filter = { <key>: <matcher>, ... };
// where <key> is a field name and
// where <matcher> is
<matcher> = <value>
        | { op: <numoperator>, val: <number> }
        | { op: <stroperator>, val: <string> }
        | { op: "and", val: <matcher>[] }
        | { op: "in", val: [<values>] }
        | { op: "notin", val: [<values>] }
        | { op: "of", val: { path: [<string>], value: <value> | <matcher>, cast?: "number" | "boolean" } }
        | { op: "oftype", val: { path: [<string>], value: <jsontype> } }
        | { op: "exists", val: <boolean> }

<numoperator> = "lte" | "lt" | "gte" | "gt"
<stroperator> = "like" | "ilike"
<jsontype>    = "object" | "array" | "string" | "number" | "boolean" | "null"
<value>       = <string> | <number> | <boolean> | null
<values>      = <value>, <values> | /* nothing */
#+END_SRC

** Transactions

Use ~dbs.transaction()~ to run multiple operations atomically:

#+BEGIN_SRC typescript
await dbs.transaction(async (t) => {
  const user = await t.collection("users").insert(
    [{ name: "Bob", email: "bob@example.com" }],
    ["id"]
  );
  
  await t.collection("profiles").insert([{
    userId: user[0].id,
    bio: "Hello!"
  }]);
});
// Automatically commits if no error, rolls back otherwise
#+END_SRC

** Error Handling

Insert and update operations throw structured error classes for common constraint violations. These classes can be imported from ~@apparts/db~ and used with ~instanceof~ checks.

| Class | Code | Meaning |
|-------|------|---------|
| ~UniqueConstraintViolation~ | ~1~ | Unique constraint violation (insert / update) |
| ~ForeignKeyConstraintViolation~ | ~2~ | Foreign key constraint violation (on delete) |
| ~CheckConstraintViolation~ | ~3~ | Check constraint violation (on insert) |

All error classes extend ~DBError~, which exposes ~code~, ~_code~, ~msg~, and ~name~.

#+BEGIN_SRC typescript
import { connect, UniqueConstraintViolation, ForeignKeyConstraintViolation, CheckConstraintViolation } from "@apparts/db";

try {
  await dbs.collection("users").insert([{ email: "duplicate@example.com" }]);
} catch (e: any) {
  if (e instanceof UniqueConstraintViolation) {
    console.log("Duplicate entry");
  }
}
#+END_SRC

For backward compatibility, the numeric ~_code~ property is still available:

#+BEGIN_SRC typescript
try {
  await dbs.collection("users").insert([{ email: "duplicate@example.com" }]);
} catch (e: any) {
  if (e._code === 1) {
    console.log("Duplicate entry");
  }
}
#+END_SRC
